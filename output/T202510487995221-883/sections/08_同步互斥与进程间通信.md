### 同步与互斥原语（锁与原子操作）

本操作系统实现了多种同步互斥原语，主要依赖 `spin` crate 提供的锁机制，并结合 Rust 标准库的原子操作实现无锁数据结构。

#### 1. UPSafeCell：单核内部可变性原语

在 `os/src/sync/up.rs` 中定义了 `UPSafeCell<T>`，用于在单核环境下安全地访问静态可变数据：

```rust
// os/src/sync/up.rs:4-32
pub struct UPSafeCell<T> {
    inner: RefCell<T>,
}

unsafe impl<T> Sync for UPSafeCell<T> {}

impl<T> UPSafeCell<T> {
    pub fn new(value: T) -> Self {
        Self { inner: RefCell::new(value) }
    }
    
    pub fn borrow_mut(&self) -> RefMut<'_, T> {
        self.inner.borrow_mut()
    }
}
```

**实现原理**：`UPSafeCell` 利用 `RefCell` 在运行时检查借用规则，通过 `unsafe impl Sync` 标记为可跨线程共享。该结构**仅适用于单核环境**，在多核场景下无法提供真正的互斥保护。

**数据结构**：

#### 2. 线程阻塞流程

当线程获取锁失败时，调用 `futex_wait_bitset` 进入阻塞：

```rust
// os/src/task/futex.rs:73-115
fn futex_wait_bitset(
    pa: usize,
    task: Arc<TaskControlBlock>,
    bitset: u32,
    timeout: Option<Timespec>,
) -> SyscallRet {
    let mut waitq = FUTEX_QUEUE_BITMAP.lock();
    let futex_key = new_futex_key();
    let waiter = FutexWaiter {
        task: Arc::downgrade(&task),
        bitset: bitset,
        futex_key: futex_key,
    };
    // 设置任务的 futex 状态
    let mut inner = task.inner_lock();
    inner.futex_pa = pa;
    inner.futex_key = futex_key;
    drop(inner);
    
    // 可选：添加超时定时器
    if let Some(timeout) = timeout {
        add_futex_timer(timeout, &task, futex_key);
    }
    
    // 插入等待队列
    if let Some(queue) = waitq.get_mut(&pa) {
        queue.push_back(waiter);
    } else {
        waitq.insert(pa, {
            let mut queue = VecDeque::new();
            queue.push_back(waiter);
            queue
        });
    }
    
    drop(waitq);
    // 阻塞当前任务并调度下一任务
    block_current_and_run_next();
    
    // 唤醒后检查是否被信号中断
    let task = current_task().unwrap();
    let task_inner = task.inner_lock();
    if !task_inner.sig_pending.difference(task_inner.sig_mask).is_empty() {
        return Err(SysErrNo::EINTR);
    }
    Ok(0)
}
```

#### 3. 线程唤醒流程

```rust
// os/src/task/futex.rs:129-164
fn futex_wake_up_bitset(pa: usize, max_num: i32, bitset: u32) -> usize {
    let mut futex_queue = FUTEX_QUEUE_BITMAP.lock();
    let mut num: usize = 0;
    if let Some(queue) = futex_queue.get_mut(&pa) {
        let queue_len = queue.len();
        let mut cnt: usize = 0;
        while cnt < queue_len && num < (max_num as usize) {
            cnt += 1;
            if let Some(waiter) = queue.pop_front() {
                if let Some(task) = waiter.task.upgrade() {
                    if bitset & waiter.bitset != 0 {
                        wakeup_futex_task(task);
                        num += 1;
                    } else {
                        queue.push_back(waiter); // 不匹配则放回
                    }
                }
            }
        }
    }
    num
}
```

```rust
// os/src/signal/mod.rs:229-244
pub fn send_signal_to_thread_group(pid: usize, sig: SigSet) -> Result<usize, SysErrNo> {
    let process = Process::get_process_arc_by_pid(pid);
    if let Some(proc) = process {
        let tasks = &proc.meta_lock().tasks;
        for task in tasks.iter() {
            if let Some(task) = task.upgrade() {
                add_signal(&task, sig);
            }
        }
        return Ok(0);
    } else {
        return Err(SysErrNo::ESRCH);
    }
}

fn add_signal(task: &TaskControlBlock, signal: SigSet) {
    let mut task_inner = task.inner_lock();
    task_inner.sig_pending |= signal;
}
```

```rust
    let timeout_opt: Option<Timespec>;
    if timeout.is_null() || !(cmd == FutexCmd::Wait || cmd == FutexCmd::WaitBitset) {
        timeout_opt = None;
    } else {
        // 转换用户态超时为内核态绝对时间
        let mut real_timeout = memory_set.get_data(timeout);
        if opt.contains(FutexOpt::FUTEX_CLOCK_REALTIME) {
            real_timeout.tv_sec -= crate::timer::NOW_TIME_STAMP;
        } else {
            real_timeout = real_timeout + get_time_spec();
        }
        timeout_opt = Some(real_timeout);
    }
    
    let val_at_uaddr = memory_set.get_data(uaddr);
    drop(memory_set);
    drop(process);
    
    match cmd {
        FutexCmd::Wait | FutexCmd::WaitBitset => {
            // 检查用户态值是否等于期望值
            if val_at_uaddr != val {
                return Err(SysErrNo::EAGAIN);
            }
            let bitset = if cmd == FutexCmd::Wait { u32::MAX } else { _val3 as u32 };
            futex_wait_bitset(pa, task, bitset, timeout_opt)
        }
        FutexCmd::Wake | FutexCmd::WakeBitset => {
            drop(task);
            let bitset = if cmd == FutexCmd::Wake { u32::MAX } else { _val3 as u32 };
            Ok(futex_wake_up_bitset(pa, val, bitset))
        }
        FutexCmd::Requeue => {
            drop(task);
            if let Some(pa2) = pa2 {
                return Ok(futex_requeue(pa, val, pa2.0, timeout as i32));
            } else {
                return Err(SysErrNo::EINVAL);
            }
        }
        _ => unimplemented!(),
    }
}
```

```rust
impl File for Pipe {
    fn read(&self, mut buf: UserBuffer) -> SyscallRet {
        assert!(self.readable());
        let mut read_size = 0usize;
        let mut loop_read;
        
        // 等待数据可用
        loop {
            let ring_buffer = self.inner_lock();
            loop_read = ring_buffer.available_read();
            if loop_read == 0 {
                if ring_buffer.all_write_ends_closed() {
                    return Ok(read_size); // EOF
                }
                drop(ring_buffer);
                suspend_current_and_run_next(); // 阻塞
                continue;
            } else {
                break;
            }
        }
        
        // 读取数据
        let mut ring_buffer = self.inner_lock();
        let length = buf.len();
        if length <= 10 {
            let mut buf_iter = buf.into_iter();
            for _ in 0..loop_read {
                if let Some(byte_ref) = buf_iter.next() {
                    unsafe { *byte_ref = ring_buffer.read_byte(); }
                    read_size += 1;
                } else {
                    break;
                }
            }
        } else {
            read_size = min(loop_read, length);
            buf.write(&ring_buffer.read_bytes(read_size));
        }
        Ok(read_size)
    }
    
    fn write(&self, mut buf: UserBuffer) -> SyscallRet {
        assert!(self.writable());
        let mut write_size = 0usize;
        let mut loop_write;
        
        // 等待缓冲区有空间
        loop {
            let ring_buffer = self.inner_lock();
            loop_write = ring_buffer.available_write();
            if loop_write == 0 {
                drop(ring_buffer);
                suspend_current_and_run_next(); // 阻塞
                continue;
            } else {
                break;
            }
        }
        
        // 写入数据
        let mut ring_buffer = self.inner_lock();
        let length = buf.len();
        if length <= 10 {
            let mut buf_iter = buf.into_iter();
            for _ in 0..loop_write {
                if let Some(byte_ref) = buf_iter.next() {
                    ring_buffer.write_byte(unsafe { *byte_ref });
                    write_size += 1;
                } else {
                    break;
                }
            }
        } else {
            write_size = min(loop_write, length);
            ring_buffer.write_bytes(&buf.read(write_size), write_size);
        }
        Ok(write_size)
    }
}
```
